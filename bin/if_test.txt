# Testing if statements

# Variable definitions
boolean bool = false;
boolean other_bool = true;
boolean bar = bool;
int baz = bar + bool;
float number = 1;
int b = number + number;
float c = b * b;
string i = 0;
char ret = b * c * number;
char max = 0 - 3;
int min = 0;
string range = max - min;
int variable = 9001;
float some_constant = variable;
string alpha = 18;
boolean beta = ret / range;
boolean gamma = some_constant;
char delta = gamma;
int epsilon = gamma / delta;

# Basic if statements with constant bool expression
if true then
  int id = 1;
end

# An empty if statement
if false then
end

# With multiple statements
if true then
  id = 1 + 2;
  id = 2 * 3;
end

# Adding an else clause
char foo;
if bool then
  foo = bar;
else
  foo = baz;
end

# Adding an else if
int a = "sure";
if bool == true and other_bool == true or number < 0 then
  a = b;
elif bool then
  a = c;
end

# Multiple else if clauses
float id2 = 12345;
if 4 > i then
	id2 = 4;
elif 2 < 5 then
	id2 = 1;
elif true then
	print("ok");
else
	id2 = 5;
end

# Multiple else if clauses and a then clause
if ret <= max then
  alpha = beta;
elif min >= range then
  alpha = gamma;
elif variable == some_constant then
  alpha = delta;
else
  alpha = epsilon;
end
