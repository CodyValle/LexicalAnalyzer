# Testing if statements

# Variable definitions
var bool = false;
var other_bool = true;
var bar = bool;
var baz = bar + bool;
var number = 1;
var b = number + number;
var c = b * b;
var i = 0;
var ret = b * c * number;
var max = 0 - 3;
var min = 0;
var range = max - min;
var variable = 9001;
var some_constant = variable;
var alpha = 18;
var beta = ret / range;
var gamma = some_constant;
var delta = gamma;
var epsilon = gamma / delta;

# Basic if statements with constant bool expression
if true then
  var id = 1;
end

# An empty if statement
if false then
end

# With multiple statements
if true then
  id = 1 + 2;
  id = 2 * 3;
end

# Adding an else clause
var foo;
if bool then
  foo = bar;
else
  foo = baz;
end

# Adding an else if
var a = "sure";
if bool == true and other_bool == true or number < 0 then
  a = b;
elif bool then
  a = c;
end

# Multiple else if clauses
var id2 = 12345;
if 4 > i then
	id2 = 4;
elif 2 < 5 then
	id2 = 1;
elif true then
	print("ok");
else
	id2 = 5;
end

# Multiple else if clauses and a then clause
if ret <= max then
  alpha = beta;
elif min >= range then
  alpha = gamma;
elif variable == some_constant then
  alpha = delta;
else
  alpha = epsilon;
end
